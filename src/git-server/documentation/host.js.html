    res = gitObject.response;

    var repo;
    this.log('Got a FETCH call for', fetch.repo);
    repo = this.getRepo(fetch.repo);
    if (repo !== false) {
      if (repo.anonRead === true) {
        this.checkTriggers('fetch', repo);
        return fetch.accept();

        return this.processSecurity(fetch, 'fetch', repo);
